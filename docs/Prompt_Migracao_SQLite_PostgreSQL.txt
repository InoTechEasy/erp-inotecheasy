TAREFA 3: CRIAR SCRIPT DE MIGRAÇÃO DE DADOS
============================================
Crie um arquivo chamado "migrar_dados.py" na raiz do projeto com o seguinte código:

---INICIO DO CODIGO---

import sqlite3
import os
from sqlalchemy import create_engine, inspect
from sqlalchemy.orm import sessionmaker

# Configuração do banco SQLite antigo (local)
sqlite_path = "banco.db"  # Altere se o nome do arquivo for diferente

# Configuração do PostgreSQL novo (Supabase)
postgres_url = "postgresql://postgres:Bancodaino%40@db.jqffbxrohqqymaeuqlqcd.supabase.co:5432/postgres"

def migrar_dados():
    """Migra dados do SQLite para PostgreSQL"""
    
    print("🔄 Iniciando migração de dados...")
    
    # Conectar ao SQLite
    try:
        sqlite_conn = sqlite3.connect(sqlite_path)
        sqlite_cursor = sqlite_conn.cursor()
        print("✅ Conectado ao SQLite")
    except Exception as e:
        print(f"❌ Erro ao conectar ao SQLite: {e}")
        return
    
    # Conectar ao PostgreSQL
    try:
        pg_engine = create_engine(postgres_url)
        pg_conn = pg_engine.connect()
        print("✅ Conectado ao PostgreSQL")
    except Exception as e:
        print(f"❌ Erro ao conectar ao PostgreSQL: {e}")
        return
    
    # Obter lista de tabelas do SQLite
    sqlite_cursor.execute("SELECT name FROM sqlite_master WHERE type='table';")
    tables = sqlite_cursor.fetchall()
    
    if not tables:
        print("⚠️  Nenhuma tabela encontrada no SQLite")
        sqlite_conn.close()
        pg_conn.close()
        return
    
    print(f"\n📊 Tabelas encontradas: {len(tables)}")
    
    # Migrar cada tabela
    for table_name in tables:
        table_name = table_name[0]
        print(f"\n🔄 Migrando tabela: {table_name}")
        
        try:
            # Obter dados da tabela SQLite
            sqlite_cursor.execute(f"SELECT * FROM {table_name}")
            rows = sqlite_cursor.fetchall()
            
            # Obter nomes das colunas
            sqlite_cursor.execute(f"PRAGMA table_info({table_name})")
            columns = [col[1] for col in sqlite_cursor.fetchall()]
            
            if not rows:
                print(f"   ⚠️  Tabela vazia: {table_name}")
                continue
            
            # Inserir dados no PostgreSQL
            for row in rows:
                placeholders = ", ".join(["%s"] * len(columns))
                column_names = ", ".join(columns)
                insert_query = f"INSERT INTO {table_name} ({column_names}) VALUES ({placeholders}) ON CONFLICT DO NOTHING"
                
                try:
                    pg_conn.execute(insert_query, row)
                except Exception as e:
                    print(f"   ⚠️  Erro ao inserir linha em {table_name}: {e}")
            
            pg_conn.commit()
            print(f"   ✅ Tabela {table_name} migrada com sucesso ({len(rows)} registros)")
        
        except Exception as e:
            print(f"   ❌ Erro ao migrar tabela {table_name}: {e}")
    
    # Fechar conexões
    sqlite_conn.close()
    pg_conn.close()
    
    print("\n✅ Migração concluída!")

if __name__ == "__main__":
    migrar_dados()

---FIM DO CODIGO---
